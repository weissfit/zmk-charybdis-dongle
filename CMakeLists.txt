execute_process(
  COMMAND git rev-parse --short=6 HEAD
  WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
  OUTPUT_VARIABLE GIT_HASH
  OUTPUT_STRIP_TRAILING_WHITESPACE
  ERROR_QUIET
)
if(NOT GIT_HASH)
  set(GIT_HASH "unknown")
endif()

set(ZMK_NAME_HASH_CONF ${CMAKE_BINARY_DIR}/zmk-name-hash.conf)
file(WRITE ${ZMK_NAME_HASH_CONF} "CONFIG_ZMK_KEYBOARD_NAME=\"Cbd-${GIT_HASH}\"\n")
set_property(GLOBAL APPEND PROPERTY EXTRA_CONF_FILE ${ZMK_NAME_HASH_CONF})
